## React 组件最小样板

```typescript
// 类型
type XxxItem = { id: number; ... };
type XxxListResponse = { items: XxxItem[]; page: number; rows_per_page: number; total: number; total_pages: number; filters: { q: string }; legacy_list_url: string };

// 组件
function XxxListPage({ apiUrl, legacyListUrl }: { apiUrl: string; legacyListUrl: string }) {
  const [q, setQ] = React.useState('')
  const [page, setPage] = React.useState(1)
  const [data, setData] = React.useState<XxxListResponse | null>(null)
  const [loading, setLoading] = React.useState(true)

  const loadList = React.useCallback(() => {
    const params = new URLSearchParams(); if (q) params.set('q', q); params.set('page', String(page)); params.set('rows_per_page', '13')
    window.history.replaceState({}, '', `${window.location.pathname}?${params.toString()}`); setLoading(true)
    fetch(`${apiUrl}?${params.toString()}`, { credentials: 'same-origin' }).then(async r => setData((await r.json()) as XxxListResponse)).finally(() => setLoading(false))
  }, [apiUrl, page, q])

  React.useEffect(() => { loadList() }, [loadList])

  return (
    <div className="react-shell-page">
      <div className="react-shell-card react-list-card">
        <div className="react-list-header">
          <div><div className="react-shell-badge">XXX（React）</div><h2>XXX管理</h2><p>已切到 React。</p></div>
          <a className="react-link-button" href={legacyListUrl}>回旧版</a>
        </div>
        <div className="react-filter-bar react-filter-bar-simple"><input value={q} onChange={e => setQ(e.target.value)} placeholder="搜索" /><button type="button" onClick={() => setPage(1)}>搜索</button></div>
        {loading ? <div className="react-shell-panel"><span>加载中...</span></div> : (
          <>
            <table className="react-table"><thead><tr>{/* columns */}</tr></thead>
              <tbody>{data?.items.map(item => <tr key={item.id}>{/* cells */}</tr>)}</tbody>
            </table>
            <div className="react-pagination-bar"><span>第 {data?.page??1} / {data?.total_pages??1} 页，共 {data?.total??0} 条</span>
              <div className="react-pagination-actions"><button disabled={(data?.page??1)<=1} onClick={()=>setPage(c=>Math.max(1,c-1))}>上一页</button><button disabled={!data||data.page>=data.total_pages} onClick={()=>setPage(c=>c+1)}>下一页</button></div></div>
          </>
        )}
      </div>
    </div>
  )
}
```

## 后端 API 样板

```python
from django.http import JsonResponse, QueryDict
from app_cybersparker import models
from app_cybersparker.utils.pagination import Pagination

ROWS_PER_PAGE_WHITELIST = {5, 10, 13, 100}

def xxx_list_api(request):
    q = request.GET.get("q", "")
    try: page = int(request.GET.get("page", "1"))
    except: page = 1
    try: rows = int(request.GET.get("rows_per_page", "13"))
    except: rows = 13
    if rows not in ROWS_PER_PAGE_WHITELIST: rows = 13

    queryset = models.Xxx.objects.all().order_by("-id")
    if q: queryset = queryset.filter(Q(name__icontains=q))

    qd = QueryDict(mutable=True); qd["page"] = str(page); qd["rows_per_page"] = str(rows)
    class _PR: GET = qd
    po = Pagination(_PR(), queryset)

    items = [{"id": o.id, ...} for o in po.page_queryset]
    return JsonResponse({"items": items, "page": po.page, "rows_per_page": po.page_size, "total": po.total_count, "total_pages": po.total_page_count, "filters": {"q": q}, "legacy_list_url": "/old/path"})
```
